#include int main() { // Declare an array int numbers[5] = {10, 20, 30, 40, 50}; // Print the elements of the array using indexing printf("Elements of the array:\n"); for (int i = 0; i < 5; i++) { printf("Index %d: %d\n", i, numbers[i]); } // Access a specific element using its index printf("\nElement at index 2: %d\n", numbers[2]); // Modify an element using its index numbers[2] = 35; printf("Modified element at index 2: %d\n", numbers[2]); return 0; } ```
